# 希望按顺序输出01234

```ts
// let
 for (let i = 0; i < 5; i++) {
    setTimeout(function() {
       console.log(i);
    }, 1000);
}

// 闭包
 for (var i = 0; i < 5; i++) {
	(function(j) {  
		setTimeout(function() {
			console.log(j);
		}, 1000);
	})(i);
}

// settimeout 第三个参数
for (var i = 0; i < 5; i++) {
	setTimeout(function(j) {
		console.log(j);
	}, 1000, i);
}

```
